     28.去除字串中重复的字符的javascript程序
    
        var s="abacabefgeeii";
         var s1=s.replace(/(.).*1/g,"$1");
         var re=new RegExp("["+s1+"]","g");
         var s2=s.replace(re,"");
         alert(s1+s2);                     //结果为：abcefgi
    
     /*使用后向引用取出包括重复的字符，再以重复的字符建立第二个表达式，取到不重复的字符，
       两者串连。这个方法对于字符顺序有要求的字符串可能不适用。*/
